66      /**
67       * Creates a connection.
68       *
69       * @param config the connection configuration to use.
70       * @param host the remote host name or IP address.
71       * @param port the remote port number.
72       * @throws LoginException if the connection configuration specifies
73       *         credentials that are either incorrect or don't provide
74       *         access to the keyspace.
75       */
76      public NodeConnection(ConnectionConfig config,
77                            String host,
78                            int port)
79              throws LoginException {
80          logger.trace("Instantiating");
81          this.host = host;
82          this.port = port;
83          this.config = config;
84          transport = new TFramedTransport(new TSocket(host, port));
85          try {
86              transport.open();
87              client = new Cassandra.Client(new TBinaryProtocol(transport));
88              if (config.getUsername() != null) {
89                  Map<String, String> creds = new HashMap<String, String>();
90                  creds.put("username", config.getUsername());
91                  creds.put("password", config.getPassword());
92                  AuthenticationRequest authRequest = new AuthenticationRequest();
93                  authRequest.setCredentials(creds);
94                  client.login(authRequest);
95              }
96          } catch (AuthenticationException e) {
97              throw new LoginException(e);
98          } catch (AuthorizationException e) {
99              throw new LoginException(e);
100         } catch (TException e) {
101             throw new RuntimeException(e);
102         }
103     }

219     @Override
220     public void addFile(String columnFamily, String key, InputStream in,
221                         Map<String, String> columns) {
222         logger.trace("Adding file: {} to column family: {}", key, columnFamily);
223         setKeyspace();
224         if (columns == null) {
225             columns = new HashMap<String, String>();
226         }
227         long timestamp = System.currentTimeMillis();
228         long byteCount = 0;
229         try {
230             // first add the chunks as rows (key = fileKey-chunk-0, data=byte[])
231             byte[] buffer = new byte[config.getFileChunkSize()];
232             int bytesRead = 0;
233             int chunkNum = 0;
234             do {
235                 bytesRead = in.read(buffer, 0, buffer.length);
236                 if (bytesRead > 0) {
237                     Column column = new Column();
238                     column.setName(buffer("bytes"));
239                     column.setValue(ByteBuffer.wrap(buffer, 0, bytesRead));
240                     column.setTimestamp(timestamp);
241                     String chunkKey = key + "-chunk-" + chunkNum;
242                     logger.trace("Adding chunk: {} to column family: {}", chunkKey, columnFamily);
243                     client.insert(buffer(chunkKey),
244                             parent(columnFamily),
245                             column,
246                             config.getWriteConsistency().getConsistencyLevel());
247                     chunkNum++;
248                     byteCount += bytesRead;
249                 }
250             } while (bytesRead >= 0);
251             // then add a row for file metadata
252             columns.put("byteCount", "" + byteCount);
253             columns.put("chunkSize", "" + config.getFileChunkSize());
254             addRecord(columnFamily, key, columns);
255         } catch (Exception e) {
256             throw new RuntimeException(e);
257         } finally{
258             IOUtils.closeQuietly(in);
259         }
260     }

262     @Override
263     public InputStream getFileContent(final String columnFamily, final String key) {
264         logger.trace("Getting file: {} from column family: {}", key, columnFamily);
265         setKeyspace();
266         final long chunkCount = getFileInfo(columnFamily, key)[2];
267         return new ChunkedInputStream(new AbstractSource<byte[]>() {
268             private long chunkNum;
269             @Override
270             protected byte[] computeNext() throws IOException {
271                 if (chunkNum == chunkCount) {
272                     return endOfData();
273                 } else {
274                     long i = chunkNum;
275                     chunkNum++;
276                     return getColumn(columnFamily,
277                                      key + "-chunk-" + i,
278                                      "bytes").getValue();
279                 }
280             }
281         });
282     }

300         long[] fileInfo = getFileInfo(columnFamily, key);

307     // [0] = byteCount, [1] = chunkSize, [2] = chunkCount
308     private long[] getFileInfo(String columnFamily, String key) {
309         Map<String, String> columns = getRecord(columnFamily, key);
310         long byteCount = Long.parseLong(columns.get("byteCount"));
311         long chunkSize = Long.parseLong(columns.get("chunkSize"));
312         long chunkCount = byteCount / chunkSize;
313         if (byteCount % chunkSize != 0) {
314             chunkCount++;
315         }
316         return new long[] { byteCount, chunkSize, chunkCount };
317     }

319     @Override
320     public void addRecord(String columnFamily, String key, Map<String, String> columns) {
321         logger.trace("Adding record: {} to column family: {}", key, columnFamily);
322         setKeyspace();
323         ByteBuffer keyBuf = buffer(key);
324         long timestamp = System.currentTimeMillis();
325         for (Map.Entry<String, String> entry: columns.entrySet()) {
326             Column column = new Column();
327             column.setName(buffer(entry.getKey()));
328             column.setValue(buffer(entry.getValue()));
329             column.setTimestamp(timestamp);
330             try {
331                 client.insert(keyBuf, parent(columnFamily), column,
332                         config.getWriteConsistency().getConsistencyLevel());
333             } catch (Exception e) {
334                 throw new RuntimeException(e);
335             }
336         }
337     }

352     private static Map<String, String> map(List<ColumnOrSuperColumn> list) {
353         Map<String, String> map = new HashMap<String, String>();
354         for (ColumnOrSuperColumn c: list) {
355             Column column = c.getColumn();
356             map.put(string(column.getName()), string(column.getValue()));

361     @Override
362     public void forEachRecord(String columnFamily, RecordFunction function) {
363         logger.trace("Iterating records in column family: {}", columnFamily);
364         setKeyspace();
365         try {
366             List<KeySlice> list;
367             String lastKey = "";
368             boolean doNext = false;
369             do {
370                 KeyRange keyRange = new KeyRange();
371                 keyRange.setStart_key(buffer(lastKey));
372                 keyRange.setEnd_key(new byte[0]);
373                 keyRange.setCount(config.getRecordBatchSize());
374                 list = client.get_range_slices(
375                         parent(columnFamily),
376                         ALL_COLUMNS,
377                         keyRange,
378                         config.getReadConsistency().getConsistencyLevel());
379                 for (KeySlice keySlice: list) {
380                     String key = string(keySlice.getKey());
381                     if (!lastKey.equals(key)) {
382                         lastKey = string(keySlice.getKey());
383                         if (lastKey.indexOf("-chunk-") == -1) { // skip file chunk rows
384                             List<ColumnOrSuperColumn> columns = keySlice.getColumns();
385                             if (columns.size() > 0) { // skip deleted rows
386                                 doNext = function.execute(lastKey, map(columns));
387                             }
388                         }
389                     }
390                 }
391             } while (list.size() > 1 && doNext);
392         } catch (Exception e) {
393             throw new RuntimeException(e);
394         }
395     }
